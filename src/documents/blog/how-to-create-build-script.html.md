<%-@partial('post-image.html.eco', { url: "how-to-create-build-script/overview-tab.jpg", title: "Overview tab" })%>

<%-@partial('post-image.html.eco', { url: "how-to-create-build-script/add-script-page.jpg", title: "Add script page" })%>
